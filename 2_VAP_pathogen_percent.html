#Sum up count for each genus

IDSeq_Taxa_VAP_MetaData.b <-  IDSeq_Taxa_VAP_MetaData %>% filter(kingdom == "<not present>") #for some reason bacteria are considered as "<not present>"

#Reduce the Matrix to Include only the IDSeq_Name, nt_count, and genus
IDSeq_Genus <- IDSeq_Taxa_VAP_MetaData.b[,c("nt_count","kingdom","genus", "sample_name")]

#Summarize by Genus for Each Sample
IDSeq_Genus_Summ_genus <- IDSeq_Genus %>% group_by(sample_name,genus) %>% summarise(nt_count_genus=sum(nt_count))

IDSeq_Genus_Summ_bact <- IDSeq_Genus %>% group_by(sample_name,kingdom) %>% summarise(nt_count_bacteria=sum(nt_count))

IDSeq_Genus_Summ <- merge(IDSeq_Genus_Summ_genus,IDSeq_Genus_Summ_bact, by="sample_name" )
IDSeq_Genus_Summ$percent_genus <- (IDSeq_Genus_Summ$nt_count_genus/IDSeq_Genus_Summ$nt_count_bacteria)*100
IDSeq_Genus_Summ <- IDSeq_Genus_Summ %>% arrange(percent_genus)
IDSeq_Genus_Summ[is.na(IDSeq_Genus_Summ)]<-0


write.csv(IDSeq_Genus_Summ,"IDSeq_byGenus_counts_sum.csv", row.names=FALSE, quote=FALSE)
