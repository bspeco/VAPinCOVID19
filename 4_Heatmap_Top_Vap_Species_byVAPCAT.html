#Add in the MetaData
MetaData <- read.csv("FINAL_FINAL_meta_microbiome_july2021.csv", header=TRUE)

MetaData_Percent_Genus <- inner_join(Data,MetaData,by="sample_name")

CDC_VAP_culture_pos = as.character(MetaData_Percent_Genus[which(MetaData_Percent_Genus["VAP_category"] == "CDC_VAP_culture_pos" &
                                                            MetaData_Percent_Genus["Early_Late"] =="Late"
                                    ),]$sample_name)
No_VAP= as.character(MetaData_Percent_Genus[which(MetaData_Percent_Genus["VAP_category"] == "No_VAP" &
                                            MetaData_Percent_Genus["Early_Late"] =="Late"
                                    ),]$sample_name)

Controls = as.character(MetaData_Percent_Genus[which(MetaData_Percent_Genus["VAP_category"] == "Control"
                                    ),]$sample_name)

MetaData_Percent_Genus = MetaData_Percent_Genus %>% filter(sample_name %in% c(CDC_VAP_culture_pos, No_VAP, Controls))


IDSeq_Genus_sample_g <- MetaData_Percent_Genus %>% group_by(genus,sample_name) %>% summarise(nt_count_genusV=sum(nt_count_genus))
IDSeq_Genus_sample_b <- MetaData_Percent_Genus %>% group_by(sample_name) %>% summarise(nt_count_bacteriaV=sum(nt_count_genus))

IDSeq_Genus_VAPCAT_g <- MetaData_Percent_Genus %>% group_by(genus,VAP_category) %>% summarise(nt_count_genusV_VAPCAT=sum(nt_count_genus))
IDSeq_Genus_VAPCAT_b <- MetaData_Percent_Genus %>% group_by(VAP_category) %>% summarise(nt_count_bacteriaV_VAPCAT=sum(nt_count_genus))

IDSeq_Genus_Summ1 <- merge(IDSeq_Genus_sample_g,IDSeq_Genus_sample_b, by=c("sample_name"), all = TRUE )
IDSeq_Genus_Summ1 <- merge(IDSeq_Genus_Summ1, MetaData_Percent_Genus, by=c("sample_name", "genus") )

IDSeq_Genus_Summ2 <- merge(IDSeq_Genus_VAPCAT_g,IDSeq_Genus_VAPCAT_b, by=c("VAP_category") )

IDSeq_Genus_Summ <- merge(IDSeq_Genus_Summ1,IDSeq_Genus_Summ2, by=c("VAP_category", "genus"))
IDSeq_Genus_Summ$percent_genus_VAPCAT <- (IDSeq_Genus_Summ$nt_count_genusV_VAPCAT/IDSeq_Genus_Summ$nt_count_bacteriaV_VAPCAT)*100

#join back with metadata

MetaData_Percent_GenusVAPCAT <- IDSeq_Genus_Summ 

TopGenuses <- c("Bacillus", "Pseudomonas","Klebsiella", "Mycoplasma", "Escherichia",
                "Prevotella","Streptococcus", "Staphylococcus", "Citrobacter", "Haemophilus", 
                "Veillonella", "Chryseobacterium", "Porphyromonas", "Corynebacterium", "Alloprevotella",
                "Lactobacillus", "Moraxella", "Neisseria", "Cutibacterium", "Fusobacterium", "Enterococcus", "Bifidobacterium")

#Filter to Include only the Top Genuses from each Sample and the Percent
library(tidyr)
IDSeq_Dom_Summ <- MetaData_Percent_GenusVAPCAT[,c("sample_name","genus","percent_genus_VAPCAT")]
IDSeq_Dom_Top <- IDSeq_Dom_Summ %>% filter(genus %in% TopGenuses)

IDSeq_Genus_Percent <- spread(IDSeq_Dom_Top,genus,percent_genus_VAPCAT)

IDSeq_Genus_Percent[is.na(IDSeq_Genus_Percent)]<-0

#Genus by Percent

IDSeq_Genus_Percent$Other <- 100-rowSums(IDSeq_Genus_Percent[,c(2:length(IDSeq_Genus_Percent))])
TopGenuses_withMetaData <- left_join(IDSeq_Genus_Percent, MetaData, by="sample_name")

write.csv(TopGenuses_withMetaData, "VAPCAT_topgenera_percent_late.csv", row.names=FALSE, quote=FALSE)

#Graph by Genus By Percent

#Library
library(tidyr)
library(ggplot2)
library(gridExtra)
Genus_Percent_R <- TopGenuses_withMetaData[,c(1:length(TopGenuses_withMetaData))]
#Convert from Wide to Long 
Genus_Percent_Long <- gather(Genus_Percent_R, Genus, Percent, Alloprevotella:Other, factor_key=TRUE)

write.csv(Genus_Percent_Long, "Heatmap_input_vapvat_late.csv", row.names=FALSE)
